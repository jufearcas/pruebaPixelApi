// Inicializar dataLayer si no existe window.dataLayer = window.dataLayer || []; document.getElementById("userForm").addEventListener("submit", function(event) { event.preventDefault(); // Prevent the actual form submission // Obtener los valores individuales del formulario const name = document.getElementById("name").value.trim(); const email = document.getElementById("email").value.trim().toLowerCase(); const phone = document.getElementById("phone").value.trim(); const phonePattern = /^\+?[1-9]\d{1,14}$/; // Validates international phone numbers if (!phonePattern.test(phone)) { alert("Por favor, ingrese un número de teléfono válido."); return; } // Hashear nombre, email y teléfono const hashedName = CryptoJS.SHA256(name).toString(); const hashedEmail = CryptoJS.SHA256(email).toString(); const hashedPhone = CryptoJS.SHA256(phone).toString(); // Enviar datos hasheados al dataLayer window.dataLayer.push({ 'event': 'formSubmission', 'formName': 'userForm', 'userName': hashedName, 'userEmail': hashedEmail, 'userPhone': hashedPhone }); // Display data in console console.log("Datos enviados al dataLayer:", { 'event': 'formSubmission', 'formName': 'userForm', 'userName': hashedName, 'userEmail': hashedEmail, 'userPhone': hashedPhone }); // Show confirmation message document.getElementById("confirmationMessage").style.display = "block"; // Limpiar el formulario después de enviar // Clear the form after submission // Si quieres que el mensaje desaparezca después de unos segundos // If you want the message to disappear after a few seconds document.getElementById("confirmationMessage").style.display = "none"; }, 5000); // El mensaje desaparecerá después de 5 segundos });
